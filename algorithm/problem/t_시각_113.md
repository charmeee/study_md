```python
import sys
limit = int(sys.stdin.readline().strip())

# h 0-23 m 0-59 s 0-59
# h> 3,13,23  60*60
# m > 3,13,23 - 53 5 60개
h = 0
m = 0
s = 0

count = 0 
for h in range(0,limit+1):
    if '3' in str(h):
        count += 60*60
        continue
    for m in range(0,60):
        if '3' in str(m):
            count += 60
            continue
        for s in range(0,60):
            if '3' in str(s):
                count += 1
                continue

print(count)

```
