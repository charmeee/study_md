### 문제
강의 1부터 n까지. 도잇에 여러강의 들을 수 있고 일부는 선수강의가 있다.
n개 강의에 대하여 수강하기까지 걸리는 최소의 시간을 각각출력

### 답
```python
from collections import deque
input = """
5
10 -1
10 1 -1
4 1 -1
4 3 1 -1
3 3 -1
"""
input = input.strip().split('\n')
n = int(input[0])
times = [0]*(n+1)
q = []
# 진입차수가 0이다. = 강의를 들어도된다.
front = [[] for _ in range(n+1)]
for i in range(1,n+1):
    t,*arr,z = list(map(int,input[i].split()))
    times[i] = t
    front[i]= arr
    if len(arr) == 0 :
        q.append(i)
        
while q:
    nq = []
    syn = [[] for _ in range(n+1)]
    # q는 진입차수 0인거
    for i in q:
        for j in range(1,n+1):
            leng = len(front[j])
            if leng != 0 and i in front[j]:
                # 들은과목 들어야하는 과목에서빼준다..
                syn[j].append(times[i])
                if leng == 1:
                    nq.append(j)
                    times[j] += max(syn[j])
                front[j].remove(i)
    q = nq
    
for i in range(1,n+1):
    print(times[i])
```
